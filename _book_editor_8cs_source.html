126 private void ProcessEvents()
127 {
128 Undo.undoRedoPerformed += UndoRedoAction;
129 if ((EventType.MouseDown == Event.current.type) && (null == dragginNode))
130 {
131 dragginNode = GetNodeAtPointer(Event.current.mousePosition + scrollPosition);
132 if(null != dragginNode)
133 {
134 dragStart = Event.current.mousePosition - dragginNode.Area.position;
135 Selection.activeObject = dragginNode;
136 }
137 else
138 {
139 draggingCanvasOffset = Event.current.mousePosition + scrollPosition;
140 draggingCanvas = true;
141 Selection.activeObject = selectedBook;
142 }
143 }
144 else if ((EventType.MouseDrag == Event.current.type) && (true == draggingCanvas))
145 {
146 scrollPosition = draggingCanvasOffset - Event.current.mousePosition;
147 GUI.changed = true;
148 }
149 else if ((EventType.MouseDrag == Event.current.type) && (null != dragginNode))
150 {
151 Rect newArea = dragginNode.Area;
152 newArea.position = Event.current.mousePosition - dragStart;
153 dragginNode.Area = newArea;
154 GUI.changed = true;
155 }
156 else if ((EventType.MouseUp == Event.current.type) && (null != dragginNode))
157 {
158 dragginNode = null;
159 }
160 else if ((EventType.MouseUp == Event.current.type) && (true == draggingCanvas))
161 {
162 draggingCanvas = false;
163 }
164 }
