100 public TType ReturnCommand<TType>(string CommandString)
101 {
102 Command seperatedCommand = Parser.Parse(CommandString);
103 string classToCall = seperatedCommand.Result;
104 if (seperatedCommand.Com == CommandRef.IS_ATTRIBUTE)
105 {
106 if (AttributPair.ContainsKey(classToCall))
107 {
108 var answer = AttributPair[classToCall].Is(seperatedCommand.Arguments);
109 return (TType)System.Convert.ChangeType(answer, typeof(TType));
110 }
111 }
112 if (seperatedCommand.Com == CommandRef.GET_INFO)
113 {
114 if (ExtraUIPair.ContainsKey(classToCall))
115 {
116 var answer = ExtraUIPair[classToCall].GetInfo<TType>();
117 return (TType)System.Convert.ChangeType(answer, typeof(TType));
118 }
119 }
120 //Debug.Log(AttributPair[seperatedCommand.Arguments[0]].Test("Is", seperatedCommand.Arguments));
121 return default(TType);
122 }
